# 최대공약수 함수 만들기

def gcdlcm(a, b):
    
    max_d = 0
    min_d = 0
    
    less = min(a, b)
    more = max(a, b)
    
    for i in range(1, less + 1):
        if less % i == 0 and more % i == 0:     # a = max(두 수), b = min(두 수), (a, a % b) -> a, b = b, a % b -> a % b = 0이거나 b > 0인 지점까지 실시
            max_d = i
    
    min_d = a * b / max_d
    
    if type(min_d) == float:
        min_d = int(min_d)
    if a == 0 or b == 0:
        print("최소공배수와 최대공약수를 구할 수 없습니다.")
    
    code = [max_d, min_d]
    
    return code
```

# 모범 답안
def gcdlcm(n, m):
    a, b = max(n, m), min(n, m)
    while b > 0:
        a, b = b, a % b
    return [a, int(n*m / a)]
```

def my_sum(n):
    print(n)
    if not n:
        return 0
    else:
        return n[0] + my_sum(n[1:])   # 리스트에서 첫 번째 요소가 빠지므로 두 번째 요소부터 다음 함수에 들어간다
